第一次前端笔试题总结——太平洋网络科技有限公司

1.正则清除空白

var str = '      fahkfakdflkafldka                ';

str = str.replace(/(^\s*)|(\s*$)/g,'');//fahkfakdflkafldka

2.http200，302,404,503 是什么意思？

200->成功

302->临时移动

404->页面找不到

503->服务器不可用

3.‘XX省xx市’如果获取分别获取省市，例如广东省广州市，如何获取广东和广州?

var str = 'XX省xx市';
var provinceIndex = str.indexOf('省');
var cityIndex = str.indexOf('市')
var province = str.slice(0,provinceIndex);
var city = str.slice(provinceIndex+1,cityIndex);
console.log(province+'-'+city);

4.如何获取333和苏州？
​
var link = [
{111:[‘广东’,’广州’]},
{222:[‘江苏’,’苏州’]},
{333:[‘广西’,’柳州’]},
{333:[‘四川’,’成都’]},
{444:[‘福建’,’厦门’]},
];
方法1：
getContent = (function(){
for (var i = 0;i<link.length;i++){
for(var key in link[i]){
if (key == 333) {
console.log(key);
}
for (var j = 0;j<link[i][key].length;j++){
if (link[i][key][j] == ‘苏州’) {
console.log(link[i][key][j]);
}
}
}
}
})();//苏州 333 333
方法2：
link.map(function(item){
for(var key in item){
if (key == 333) {
console.log(key)
}
item[key].map(function(item2){
if (item2 == ‘苏州’) {
console.log(item2);
}
})
}//苏州 333 333
})

5.数组去重
​
Array.prototype.noRepeat = function(){
var arr = [];
var obj = {};
for(var i = 0;i<this.length;i++){
if (!obj[this[i]]) {
arr.push(this[i]);
obj[this[i]]= this[i];
}
}
return arr;
}
myarr = [1,2,4,1,5,6,1,7,8,9]
myarr.noRepeat();//[1, 2, 4, 5, 6, 7, 8, 9]
